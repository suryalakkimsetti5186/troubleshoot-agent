// FUNCTION TO ACTUALLY OPEN APPLICATIONS
// ============================================
function openApplication(appName, res) {
    console.log(`🔧 Attempting to open: "${appName}"`);
    
    // Common app mappings
    const appMappings = {
        'chrome': 'chrome',
        'google chrome': 'chrome',
        'firefox': 'firefox',
        'edge': 'msedge',
        'notepad': 'notepad',
        'calculator': 'calc',
        'calc': 'calc',
        'paint': 'mspaint',
        'cmd': 'cmd',
        'command prompt': 'cmd',
        'powershell': 'powershell',
        'terminal': 'cmd',
        'spotify': 'spotify',
        'vscode': 'code',
        'visual studio code': 'code',
        'word': 'winword',
        'excel': 'excel',
        'powerpoint': 'powerpnt',
        'outlook': 'outlook',
        'task manager': 'taskmgr',
        'control panel': 'control',
        'explorer': 'explorer',
        'file explorer': 'explorer',
        'this pc': 'explorer',
        'recycle bin': 'start shell:RecycleBinFolder',
        'settings': 'start ms-settings:',
    };
    
    // Find matching app
    let commandToRun = null;
    const lowerApp = appName.toLowerCase();
    
    for (const [key, value] of Object.entries(appMappings)) {
        if (lowerApp.includes(key) || key.includes(lowerApp)) {
            commandToRun = value;
            break;
        }
    }
    
    // If no mapping found, try the name directly
    if (!commandToRun) {
        commandToRun = appName;
    }
    
    // Execute the command
    const fullCommand = `start ${commandToRun}`;
    console.log(`📂 Running: ${fullCommand}`);
    
    exec(fullCommand, (error, stdout, stderr) => {
        if (error) {
            console.error(`❌ Error opening ${appName}:`, error.message);
            // Try alternative method
            exec(`start "" "${appName}"`, (err2) => {
                if (err2) {
                    res.json({ success: false, message: `Could not open "${appName}". Please check if it's installed.` });
                } else {
                    res.json({ success: true, message: `✅ Opened ${appName}` });
                }
            });
        } else {
            console.log(`✅ Successfully opened: ${appName}`);
            res.json({ success: true, message: `✅ Opened ${appName}` });
        }
    });
}

// FUNCTION TO ACTUALLY CONTROL BLUETOOTH
// ============================================
function controlBluetooth(action, res) {
    console.log(`🔧 Bluetooth: ${action}`);
    
    const psCommand = `
        $btAdapter = Get-PnpDevice -Class Bluetooth | Where-Object { $_.FriendlyName -like "*Bluetooth*" -or $_.Class -eq "Bluetooth" }
        if ($btAdapter) {
            if ("${action}" -eq "off") {
                Disable-PnpDevice -InstanceId $btAdapter.InstanceId -Confirm:$false
                Write-Output "SUCCESS: Bluetooth turned OFF"
            } else {
                Enable-PnpDevice -InstanceId $btAdapter.InstanceId -Confirm:$false
                Write-Output "SUCCESS: Bluetooth turned ON"
            }
        } else {
            Write-Output "ERROR: Bluetooth adapter not found"
        }
    `;
    
    exec(`powershell -Command "${psCommand.replace(/"/g, '\\"')}"`, (error, stdout) => {
        if (error || stdout.includes('ERROR')) {
            console.log('PowerShell method failed, opening settings...');
            exec('start ms-settings:bluetooth', () => {
                res.json({ success: true, message: 'Opening Bluetooth Settings' });
            });
        } else {
            console.log(`✅ Bluetooth ${action}`);
            res.json({ success: true, message: stdout.trim() });
        }
    });
}

// ============================================
// FUNCTION TO CONTROL WIFI
// ============================================
function controlWiFi(action, res) {
    console.log(`🔧 WiFi: ${action}`);
    
    const psCommand = `
        $wifiAdapter = Get-NetAdapter | Where-Object { $_.Name -like "*WiFi*" -or $_.Name -like "*Wireless*" }
        if ($wifiAdapter) {
            if ("${action}" -eq "off") {
                Disable-NetAdapter -Name $wifiAdapter.Name -Confirm:$false
                Write-Output "SUCCESS: WiFi turned OFF"
            } else {
                Enable-NetAdapter -Name $wifiAdapter.Name -Confirm:$false
                Write-Output "SUCCESS: WiFi turned ON"
            }
        } else {
            Write-Output "ERROR: WiFi adapter not found"
        }
    `;
    
    exec(`powershell -Command "${psCommand.replace(/"/g, '\\"')}"`, (error, stdout) => {
        if (error || stdout.includes('ERROR')) {
            exec('start ms-settings:network-wifi', () => {
                res.json({ success: true, message: 'Opening WiFi Settings' });
            });
        } else {
            console.log(`✅ WiFi ${action}`);
            res.json({ success: true, message: stdout.trim() });
        }
    });
}

// SYSTEM CONTROL FUNCTIONS
// ============================================
function lockScreen(res) {
    console.log('🔒 Locking screen');
    exec('rundll32.exe user32.dll,LockWorkStation', (error) => {
        if (error) {
            res.json({ success: false, message: 'Could not lock screen' });
        } else {
            res.json({ success: true, message: '🔒 Screen locked' });
        }
    });
}

function sleepComputer(res) {
    console.log('💤 Putting computer to sleep');
    exec('rundll32.exe powrprof.dll,SetSuspendState 0,1,0', (error) => {
        if (error) {
            res.json({ success: false, message: 'Could not sleep' });
        } else {
            res.json({ success: true, message: '💤 Going to sleep' });
        }
    });
}

function shutdownComputer(res) {
    console.log('🖥️ Shutting down');
    exec('shutdown /s /t 30', (error) => {
        if (error) {
            res.json({ success: false, message: 'Could not shutdown' });
        } else {
            res.json({ success: true, message: '🖥️ Shutting down in 30 seconds' });
        }
    });
}

function restartComputer(res) {
    console.log('🔄 Restarting');
    exec('shutdown /r /t 30', (error) => {
        if (error) {
            res.json({ success: false, message: 'Could not restart' });
        } else {
            res.json({ success: true, message: '🔄 Restarting in 30 seconds' });
        }
    });
}

function cancelShutdown(res) {
    console.log('❌ Cancelling shutdown');
    exec('shutdown /a', (error) => {
        if (error) {
            res.json({ success: false, message: 'No shutdown in progress' });
        } else {
            res.json({ success: true, message: '❌ Shutdown cancelled' });
        }
    });
}

// SYSTEM INFO FUNCTIONS
// ============================================
async function getCPUInfo(res) {
    try {
        const cpu = await si.cpu();
        const load = await si.currentLoad();
        const temp = await si.cpuTemperature();
        
        const message = `📊 CPU: ${cpu.brand}\n📈 Usage: ${load.currentLoad.toFixed(1)}%\n🌡️ Temperature: ${temp.main ? temp.main.toFixed(1) + '°C' : 'N/A'}\n⚡ Cores: ${cpu.cores}`;
        res.json({ success: true, message: message });
    } catch (error) {
        res.json({ success: false, message: 'Could not get CPU info' });
    }
}

async function getMemoryInfo(res) {
    try {
        const mem = await si.mem();
        const usage = ((mem.active / mem.total) * 100).toFixed(1);
        const message = `💾 RAM Usage: ${usage}%\n📊 Used: ${(mem.active / 1073741824).toFixed(2)}GB / ${(mem.total / 1073741824).toFixed(2)}GB`;
        res.json({ success: true, message: message });
    } catch (error) {
        res.json({ success: false, message: 'Could not get memory info' });
    }
}

async function getDiskInfo(res) {
    try {
        const disks = await si.fsSize();
        let message = `💿 DISK STATUS\n`;
        for (const disk of disks) {
            const usage = ((disk.used / disk.size) * 100).toFixed(1);
            message += `\n📀 ${disk.mount}:\n   Used: ${usage}% (${(disk.used / 1073741824).toFixed(2)}GB / ${(disk.size / 1073741824).toFixed(2)}GB)`;
        }
        res.json({ success: true, message: message });
    } catch (error) {
        res.json({ success: false, message: 'Could not get disk info' });
    }
}

app.post('/api/voice-command', async (req, res) => {
    const { command } = req.body;
    const lower = command.toLowerCase();
    console.log(`\n🎤 COMMAND: "${command}"`);
    
    // Store command in database
    db.run(`INSERT INTO commands (command) VALUES (?)`, [command]);
    
    // ========== OPEN APPLICATIONS ==========
    if (lower.includes('open ') || lower.startsWith('open')) {
        let appName = command.replace(/open /i, '').trim();
        openApplication(appName, res);
        return;
    }
    
    // ========== BLUETOOTH ==========
    if (lower.includes('bluetooth')) {
        if (lower.includes('off') || lower.includes('disable') || lower.includes('turnoff')) {
            controlBluetooth('off', res);
        } else if (lower.includes('on') || lower.includes('enable') || lower.includes('turnon')) {
            controlBluetooth('on', res);
        } else {
            res.json({ success: true, message: 'Say "turn off bluetooth" or "turn on bluetooth"' });
        }
        return;
    }
    
    // ========== WIFI ==========
    if (lower.includes('wifi') || lower.includes('wi-fi')) {
        if (lower.includes('off') || lower.includes('disable')) {
            controlWiFi('off', res);
        } else if (lower.includes('on') || lower.includes('enable')) {
            controlWiFi('on', res);
        } else {
            res.json({ success: true, message: 'Say "disable wifi" or "enable wifi"' });
        }
        return;
    }
    
    // ========== LOCK SCREEN ==========
    if (lower.includes('lock')) {
        lockScreen(res);
        return;
    }
    
    // ========== SLEEP ==========
    if (lower.includes('sleep')) {
        sleepComputer(res);
        return;
    }
    
    // ========== SHUTDOWN / RESTART ==========
    if (lower.includes('shutdown') || lower.includes('shut down')) {
        if (lower.includes('cancel')) {
            cancelShutdown(res);
        } else {
            shutdownComputer(res);
        }
        return;
    }
    
    if (lower.includes('restart')) {
        if (lower.includes('cancel')) {
            cancelShutdown(res);
        } else {
            restartComputer(res);
        }
        return;
    }
    
    // ========== VOLUME CONTROL ==========
    if (lower.includes('volume')) {
        if (lower.includes('up')) {
            volumeUp(res);
        } else if (lower.includes('down')) {
            volumeDown(res);
        } else if (lower.includes('mute')) {
            mute(res);
        } else {
            res.json({ success: true, message: 'Say "volume up", "volume down", or "mute"' });
        }
        return;
    }
    
    // ========== SYSTEM INFO ==========
    if (lower.includes('cpu')) {
        await getCPUInfo(res);
        return;
    }
    
    if (lower.includes('memory') || lower.includes('ram')) {
        await getMemoryInfo(res);
        return;
    }
    
    if (lower.includes('disk') || lower.includes('storage')) {
        await getDiskInfo(res);
        return;
    }
    
    // ========== CIVIC ISSUES ==========
    if (lower.includes('streetlight') || (lower.includes('light') && lower.includes('not working'))) {
        res.json({ success: true, message: '✅ Streetlight issue reported. Work order #ST-2401 created.' });
        return;
    }
    
    if (lower.includes('traffic') && (lower.includes('signal') || lower.includes('light'))) {
        res.json({ success: true, message: '✅ Traffic signal issue reported. Control room notified.' });
        return;
    }
    
    if (lower.includes('water') && (lower.includes('leak') || lower.includes('leakage'))) {
        res.json({ success: true, message: '✅ Water leakage reported. Emergency team dispatched.' });
        return;
    }
    
    if (lower.includes('garbage') || lower.includes('waste')) {
        res.json({ success: true, message: '✅ Waste collection requested.' });
        return;
    }
    
    if (lower.includes('pothole')) {
        res.json({ success: true, message: '✅ Pothole reported. Road repair team notified.' });
        return;
    }
    
    // ========== HELP ==========
    if (lower.includes('help')) {
        res.json({ success: true, message: `📋 AVAILABLE COMMANDS:

🔹 OPEN APPS: "open chrome", "open notepad", "open calculator"
🔹 BLUETOOTH: "turn off bluetooth", "turn on bluetooth"
🔹 SYSTEM: "lock screen", "sleep", "shutdown", "restart"
🔹 INFO: "check CPU", "check memory", "disk status"
🔹 VOLUME: "volume up", "volume down", "mute"
🔹 CIVIC: "streetlight not working", "water leakage"` });
        return;
    }
    
    // ========== DEFAULT ==========
    res.json({ success: true, message: `I heard: "${command}". Try "help" to see available commands.` });
});

            ram_used: (mem.active / 1073741824).toFixed(2),

        };
        res.json({ success: true, metrics, status: 'healthy' });
    } catch (error) {
        res.json({ success: false, error: error.message });
    }
});
